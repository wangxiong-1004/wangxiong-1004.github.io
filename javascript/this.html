# 理解this

# 不同执行环境下的this

# 1. 全局环境下执行

console.log(this) // window

在全局作用域中 this 指向当前的全局对象 Window。

# 2. 函数内部执行

function func() {
  console.log(this)
}

func() // window

# 3. 作为对象到方法调用

var obj = {
  name: 'wang',
  showName: function() {
    console.log(this)
  }
}
obj.showName() // obj
var showName = obj.showName
showName() // window
  • 对象直接调用,则指向对象
  • 把对象到方法赋值给一个全局变量后,则指向window

# 4. 作为一个构造函数使用

function Person(name) {
  this.name = name
  console.log(this)
}

var person = new Person('wang')

this指向实例化后的对象

# 5. 在定时器中使用

setTimeout(function() {
  console.log(this)
}, 1000)

指向window

# 6. 箭头函数

箭头函数中 this 的值取决于该函数外部非箭头函数的 this 的值，且不能通过 call() 、 apply() 和 bind() 方法来改变 this 的值。

var func = () => {
	console.log(this)
}

func() // window

var obj = {
	name: 'wang',
  showName: function() {
    console.log(this)
  }
}
obj.showName() // obj

var obj = {
	name: 'wang',
  showName: () => {
    console.log(this)
  }
}
obj.showName() // window

# 7.call / apply / bind

# call

fun.call(thisArg[, arg1[, arg2[, ...]]])

它会立即执行函数，第一个参数是指定执行函数中 this 的上下文，后面的参数是执行函数需要传入的参数。

# apply

fun.apply(thisArg, [argsArray])

它也会立即执行函数，第一个参数是指定执行函数中 this 的上下文，第二个参数是一个数组，是传给执行函数的参数（与 call 的区别）。

# bind

var foo = fun.bind(thisArg[, arg1[, arg2[, ...]]]);

它不会执行函数，而是返回一个新的函数，这个新的函数被指定了 this 的上下文，后面的参数是执行函数需要传入的参数。

# 如何改变this指向

# 1.使用es6的箭头函数

var obj = {
	showName: function() {
    setTimeout(function() { // 普通函数
      console.log(this)
    }, 1000)
  }
}
obj.showName() // window

var obj = {
	showName: function() {
    setTimeout(() => { // 箭头函数
      console.log(this)
    }, 1000)
  }
}
obj.showName() // obj

# 2.内部保存this

var obj = {
	showName: function() {
		var _this = this
    setTimeout(function() {
      console.log(_this)
    }, 1000)
  }
}
obj.showName() // obj

# 3.使用call/apply/bind

call:

var obj = {
	showName: function() {
    setTimeout(function() {
      console.log(this)
    }.call(obj), 1000)
  }
}
obj.showName() // obj

apply:

var obj = {
	showName: function() {
    setTimeout(function() {
      console.log(this)
    }.apply(obj), 1000)
  }
}
obj.showName() // obj

bind:

var obj = {
	showName: function() {
    setTimeout(function() {
      console.log(this)
    }.bind(obj), 1000)
  }
}
obj.showName() // obj
